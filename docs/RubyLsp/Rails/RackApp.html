def call(env)
  request = ActionDispatch::Request.new(env)
  path = request.path

  route, argument = path.delete_prefix(BASE_PATH).split("/")

  case route
  when "activate"
    [200, { "Content-Type" => "application/json" }, []]
  when "models"
    resolve_database_info_from_model(argument)
  else
    not_found
  end
end
